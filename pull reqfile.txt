#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void) 
{
    FILE* fp;
    int n, i;

    fp = fopen("data.txt", "w");
    if (fp == NULL) 
    {
        printf("파일을 열 수 없다.\n");
        return 1;
    }

    printf("저장할 정수 개수 입력: ");
    scanf("%d", &n);

    srand(time(NULL));

    for (i = 0; i < n; i++) 
    {
        int num = rand() % 100 + 1;
        fprintf(fp, "%d\n", num);
    }

    fclose(fp);

    printf("*** 저장이 완료되었습니다.\n");

    return 0;
}
